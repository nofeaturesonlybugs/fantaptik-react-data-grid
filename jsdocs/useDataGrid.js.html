useDataGrid.js

import React from 'react';

import useData, { useDataContext} from './useData';
import usePages, { usePagesContext } from './usePages';

/**
 * The result type for the `useDataGrid` hook.
 * 
 * @typedef useDataGridResult
 * @type {Object}
 * @property {any[]} data The data rows.
 * @property {useDataAppendData} appendData Appends data to the `data` value.
 * @property {useDataSetData} setData Sets the `data` value.
 * @property {usePagesResult} pages The pages data.
 */

/**
 * `useDataGrid` creates a data grid management object.
 * 
 * @param {Object} props The initial context values.
 * @param {any[]} [props.data=[]] The initial data. 
 * @returns {useDataGridResult}
 */
const useDataGrid = ( { data : dataDefault = [] } = {} ) => {
    const [__ctx] = React.useState( new Date().toString() );
    const data = useData( { data : dataDefault } );
    const pages = usePages();
    //
    React.useEffect( () => {
        pages.setItemCount( data.data.length );
    }, [data.data] );
    //
    return {
        __ctx,
        ...data,
        pages,
    };
}

export const useDataGridContext = {
    __ctx : null,
    ...useDataContext,
    pages : { ...usePagesContext },
}

export default useDataGrid;
